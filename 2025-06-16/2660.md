# [2660 회장뽑기](https://www.acmicpc.net/problem/2660)

```python
import sys
input = sys.stdin.readline

# 노드의 개수
n = int(input())
graph = [[50] * (n + 1) for _ in range(n + 1)]
distance = [0] * (n + 1)

# 자기 자신에서 자기 자신으로 가는 비용을 0으로 초기화
for i in range(1, n + 1):
    for j in range(1, n + 1):
        if i == j:
            graph[i][j] = 0

# 간선 입력
while True:
    a, b = map(int, input().split())

    if a == -1 and b == -1:
        break

    # 연결되어 있다면, 1 (친구이면 1)
    graph[a][b] = 1
    graph[b][a] = 1

# 점화식에 따라 플로이드 워셜 알고리즘 수행
for i in range(1, n + 1):
    for a in range(1, n + 1):
        for b in range(1, n + 1):
            graph[a][b] = min(graph[a][b], graph[a][i] + graph[i][b])

for i in range(1, n + 1):
    distance[i] = max(graph[i][1:])

t = min(distance[1:]) # 점수가 가장 적은 사람이 회장
print(t, distance.count(t)) # count로 회장의 수를 센다
for i in range(1, n + 1):
    if distance[i] == t:
        print(i, end=' ')
```
